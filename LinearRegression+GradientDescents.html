import math as math

""" 
t vn :
Gi s ta c X l cc c im ca mt cn nh v Y l cc gi tr c ngi nh  khi  c cho bit trc
liu ta c th d on c gi c ca ngi nh khi ta c mt X bt k?
Chc chn l c. ta s thy c mi quan h tuyn tnh gia c im ca mt ngi nh vi gi c ca ngi nh 
"""

# khi to d liu
X = np.array([10, 22, 31, 42, 54, 72 , 91])  #din tch ca mt ngi nh ( bin u vo X )
y = np.array([1.5, 3.2, 4.8, 6.4, 8.1 ,11, 13 ])   #gi ca mt ngi nh  ( bin u ra Y )

"""
Bias trong Machine Learning c th hiu l s thin v, n th hin s sai lch ca kt qu d on so vi gi tr thc t ca m hnh
VD: Nu m hnh lun d on gi nh thp hn gi tr thc t 10%, th m hnh ny c bias m v ngc 
"""
# Thm ct 1 vo X  i din cho h s chn (bias)
X_b = np.c_[np.ones((len(X), 1)), X]  # To ma trn (n, 2)

# Tnh ton cc h s theta s dng cng thc  = (X^T * X)^-1 * X^T * y
theta_best = np.linalg.inv(X_b.T.dot(X_b)).dot(X_b.T).dot(y)

"""
Nu khng mun gii tch, c th s dng Gradient Descent: lp li dn  ti u theta
"""
# Khi to gradients
theta = np.zeros(X_b.shape[1])
learning_rate =0.0001
iterations = 100
m = len(y)
# GDescentsDescents
for _ in range(iterations):
    gradients = (1/m) * X_b.T.dot(X_b.dot(theta) - y)
    theta -= learning_rate * gradients
# D on
y_pred = X_b.dot(theta_best)
y_gradients_pred = X_b.dot(theta)

# Hin th kt qu
print("H s theta:", theta_best,)
print("H s theta theo gradient descents l:" ,theta)
print("D on  gi tr ca ngi nh s :", y_pred, )
print("D on  gi tr ca ngi nh s :", y_gradients_pred, )

def calculate_mae(y_true, y_pred):
    return np.mean(np.abs(y_true - y_pred))
def calculate_mse(y_true, y_pred):
    return np.mean((y_true - y_pred) ** 2)
